// Open the iphone simulator. Must have Xcode installed and be on a Mac
open -a simulator

// Select hardware in toolbar. We select iphone 11 pro Max

// Set up 'show touch indicator' on the iphone simulator
defaults write com.apple.iphonesimulator ShowSingleTouches 1

// Record video on the simulator
xcrun simctl io booted recordVideo <filename>.<extension>

// Our recording path and name
xcrun simctl io booted recordVideo ../Recordings/iphone_01.mp4

// Take screenshot on the simulator
xcrun simctl io booted screenshot
